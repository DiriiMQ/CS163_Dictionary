168void SearchBox::updateText() {
169 int codepointSize = 0, totalCodepointSize = 0, charCounter = 0, last = (int)this->rawText.length();
170 const char *raw = this->rawText.c_str();
171 std::vector<std::string> result;
172 for (; last; last -= codepointSize, totalCodepointSize += codepointSize, charCounter++) {
173 result.clear();
174 Utils::formatString(this->font, 1, this->rawText.substr(last, totalCodepointSize), MAX_LENGTH_PIXEL, this->fontSize, result);
175 if (result.size() > 1) {
176 break;
177 }
178 codepointSize = 0;
179 GetCodepointPrevious(raw + last, &codepointSize);
180 }
181 this->text = this->rawText.substr(last, totalCodepointSize);
182}
